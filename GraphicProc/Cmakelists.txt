cmake_minimum_required(VERSION 3.16)
project(OpenGL_Scene VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# ── Find required packages ────────────────────────────────────────────────────
find_package(OpenGL REQUIRED)

# GLFW
find_package(glfw3 QUIET)
if(NOT glfw3_FOUND)
    message(STATUS "glfw3 not found via find_package – trying pkg-config")
    find_package(PkgConfig REQUIRED)
    pkg_check_modules(GLFW REQUIRED glfw3)
endif()

# GLM (header-only, no link needed)
find_package(glm QUIET)

# ── Source files ──────────────────────────────────────────────────────────────
# GLAD is included directly as a single .c file (see glad/ folder)
set(SOURCES
    main.cpp
    glad/glad.c          # include glad source here
)

add_executable(${PROJECT_NAME} ${SOURCES})

# ── Include paths ─────────────────────────────────────────────────────────────
target_include_directories(${PROJECT_NAME} PRIVATE
    ${CMAKE_SOURCE_DIR}/include   # place glad/glad.h and KHR/khrplatform.h here
    ${GLFW_INCLUDE_DIRS}
)

# ── Link libraries ────────────────────────────────────────────────────────────
target_link_libraries(${PROJECT_NAME}
    OpenGL::GL
    glfw
    ${CMAKE_DL_LIBS}          # required for GLAD on Linux (-ldl)
)

# Windows also needs these
if(WIN32)
    target_link_libraries(${PROJECT_NAME} opengl32)
endif()

# macOS needs these frameworks
if(APPLE)
    target_link_libraries(${PROJECT_NAME}
        "-framework OpenGL"
        "-framework Cocoa"
        "-framework IOKit"
        "-framework CoreVideo"
    )
endif()

message(STATUS "
  ╔══════════════════════════════════════════════╗
  ║   OpenGL Scene – World in a Cube             ║
  ║   Build with:  cmake -B build && cmake --build build ║
  ╚══════════════════════════════════════════════╝
")
